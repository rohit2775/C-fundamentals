 
  ## ove4rloading of unary opereator  &&&& pre and post increment operator 

    #include <iostream>
using namespace std;

class Complex {
private:
    int a, b;

public:
    void setData(int x, int y) {
        a = x;
        b = y;
    }

    void showData() {
        cout << a << " " << b << endl;
    }

    // Overloading unary operator (-)
    Complex operator-() {
        Complex temp;
        temp.a = -a;
        temp.b = -b;
        return temp;
    }
};

int main() {
    Complex c1, c2;
    c1.setData(3, 4);

    cout << "Original c1: ";
    c1.showData();

    // Using operator function explicitly
    c2 = c1.operator-();
    cout << "After overloading (explicit call): ";
    c2.showData();

    // OR using shorthand
    c2 = -c1;
    cout << "After overloading (using -c1): ";
    c2.showData();

    return 0;
}
